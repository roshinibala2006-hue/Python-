a = 29
b =30
data = {
    "a": a,
    "b": b,
    "Total Students": a + b,
    "Absentees": b,
    "Present": a
}
for key in data:
    print(f"{key}: {data[key]}")

if data['Present'] + data['Absentees'] == data['Total Students']:
    print("Attendance is correct.")
elif data['Present'] + data['Absentees'] > data['Total Students']:
    print("Attendance is wrong.")
else:
    print("Attendance not fully marked.")

output:
a: 29
b: 30
Total Students: 59
Absentees: 30
Present: 29
Attendance is correct.

def check_login():
    for i in range(1, 4):
        if input(f"Attempt {i}: Enter your password: ") == "1234":
            print(f"Login successful on attempt {i}")
            break
        print("Wrong password, try again.")
    else:
        print("Access denied. Too many failed attempts.")
check_login()
